package AutomotiveHardwareDefinitions {

    part def CameraSensor {
        attribute name : String;
        attribute resolution : String;
        attribute fov : Real;
    }

    part def LidarSensor {
        attribute name : String;
        attribute resolution : String;
        attribute fov : Real;
        attribute range : Real;
    }

    part def RadarSensor {
        attribute name : String;
        attribute resolution : String;
        attribute fov : Real;
        attribute range : Real;
    }

    part def BrakeActuator {
        attribute name : String;
        attribute brakingLevel : Real;
    }

    part def SteeringActuator {
        attribute name : String;
        attribute steeringAngle : Real;
    }

    part def HardwarePlatform {
        attribute name : String;

        part camera : CameraSensor;
        part lidar : LidarSensor;
        part radar : RadarSensor;
        part brakes : BrakeActuator;
        part steering : SteeringActuator;
    }
}
